// creditScore(iso): calcula la media de los scores disponibles de las 3 agencias
// Devuelve 0-100 o null si no hay ningún rating.
function creditScore(iso){
  const r=CREDIT_RATINGS[iso];
  if(!r)return null;
  const parts=[ratingToScore(r.SP),ratingToScore(r.Moody),ratingToScore(r.Fitch)]
    .filter(v=>v!=null);
  return parts.length?Math.round(parts.reduce((a,b)=>a+b,0)/parts.length):null;
}

// GDELT — CONVERSIÓN A KPI LEGIBLE
// ══════════════════════════════════════════════════════════════
// GDELT mide "inestabilidad" en un rango 0.00 - 0.20+
// (más alto = más inestable, más difícil de interpretar).
// Lo convertimos a un KPI 0-100 donde 100 = máxima estabilidad.
//
// gdeltToKPI(raw): KPI = max(0, 100 - raw × 500)
//   Ejemplo: Ucrania raw=0.172 → KPI=14  → Crisis severa
//   Ejemplo: España  raw=0.008 → KPI=96  → Muy estable
//
// gdeltLabel(kpi): devuelve emoji + texto + color según el KPI
//   para mostrarlo en tooltip, tabla y modal de forma consistente.
// ══════════════════════════════════════════════════════════════
function gdeltToKPI(raw){return raw==null?null:Math.max(0,Math.round(100-raw*500));}
function gdeltLabel(kpi){
  if(kpi==null)return{emoji:'⬜',label:'Sin datos',sub:'No hay datos GDELT',bg:'#f1f5f9',color:'#475569'};

  if(kpi>=60)return{emoji:'🟠',label:'Tensión activa',sub:'Conflictividad por encima de lo normal',bg:'#fff7ed',color:'#9a3412'};
  if(kpi>=40)return{emoji:'🔴',label:'Inestabilidad significativa',sub:'Conflictos o crisis con impacto real',bg:'#fef2f2',color:'#991b1b'};

}

// FÓRMULA DE SCORING — EL CORAZÓN DE LA APP
// ══════════════════════════════════════════════════════════════
// buildScore(iso): calcula la puntuación 0-100 de un país.
//
// PASO 1 — Normalización
//   Cada variable viene en unidades distintas (dólares, %, índice...).
//   Las convertimos todas a escala 0-100 para poder combinarlas.
//   wn(v)  = convierte WGI (-2.5 a +2.5) → 0-100
//   nr(v)  = convierte cualquier variable con min/max → 0-100
//            inv=true significa que más bajo es mejor (ej: desempleo)
//
// PASO 2 — Índice Social compuesto
//   Social = Libertades×0.38 + CapHumano×0.25 + Empleo×0.20 + Pobreza×0.17
//   Solo se calcula con las partes disponibles (si falta alguna,
//   se redistribuye el peso entre las demás).
//
// PASO 3 — CountryScore (media ponderada)
//   CountryScore = Σ(variable × peso) / Σ(pesos disponibles)
//   Si un dato falta, ese peso se reparte entre los demás.
//   Pesos: PoliticalStability=0.20, Governance=0.10, Corruption=0.10...
//
// PASO 4 — Penalización GDELT
//   Si GDELT detecta conflicto activo, el score se reduce:
//   raw > 0.08 → ×0.40  (guerra abierta, score se reduce al 40%)
//   raw > 0.04 → ×0.65  (conflicto activo)
//   raw > 0.02 → ×0.85  (tensión)
//
// RESULTADO:
//   CountryScore ≥ 65 → Riesgo BAJO
//   CountryScore 40-64 → Riesgo MEDIO
//   CountryScore < 40 → Riesgo ALTO
//   isConflict=true → etiqueta especial ⚠ CONFLICTO
// ══════════════════════════════════════════════════════════════
function wn(v){return v==null?null:Math.round(((v+2.5)/5)*100);}
function nr(v,mn,mx,inv=false){if(v==null)return null;let p=(Math.max(mn,Math.min(mx,v))-mn)/(mx-mn)*100;return Math.round(inv?100-p:p);}

function buildScore(iso){
  const w=WGI[iso]||{},d=WDI[iso]||{};
  const PS=wn(w.PV),GV=wn(w.GE),CC=wn(w.CC),RL=wn(w.RL),VA=wn(w.VA),RQ=wn(w.RQ);
  const FN=nr(d.GDP,500,80000),EL=nr(d.EL,0,100),EN=nr(d.CO2,0,20,true),BIO=nr(d.FR,0,100),WC=nr(d.UN,0,35,true),LIT=nr(d.LIT,20,100),POV=nr(d.POV,0,80,true);
  const EV=EN!=null&&BIO!=null?Math.round((EN+BIO)/2):EN??BIO;
  const COR=CPI[iso]??CC;
  const CL=FH[iso]!=null?Math.round((7-FH[iso])/6*100):null;
  const CR=creditScore(iso); // Rating crediticio S&P + Moody's + Fitch (0-100)
  const socParts=[[CL,.38],[LIT,.25],[WC,.20],[POV,.17]].filter(([v])=>v!=null);
  const SOC=socParts.length?Math.round(socParts.reduce((a,[v,wt])=>a+v*wt,0)/socParts.reduce((a,[,wt])=>a+wt,0)):null;
  // Pesos actualizados: CreditRating tiene peso 0.08 (refleja solvencia financiera soberana)
  // Se reduce Financial de 0.08 a 0.05 y Environmental de 0.03 a 0.02 para compensar
  const parts=[[PS,.20],[GV,.10],[COR,.10],[RL,.08],[CL,.08],[VA,.06],[RQ,.06],[FN,.05],[CR,.08],[LIT,.05],[EL,.04],[WC,.04],[POV,.04],[EN,.02],[BIO,.02],[EV,.02]].filter(([v])=>v!=null);
  let sc=0,wt=0;parts.forEach(([v,pw])=>{sc+=v*pw;wt+=pw;});
  if(wt<0.25)return null;
  let cs=Math.round(sc/wt);
  const g=GDELT_DATA[iso]??null;
  if(g!=null){if(g>0.08)cs=Math.round(cs*.40);else if(g>0.04)cs=Math.round(cs*.65);else if(g>0.02)cs=Math.round(cs*.85);}
  cs=Math.max(0,Math.min(100,cs));
  const isCon=CONF.has(iso)||(g!=null&&g>0.04)||(cs<20);
  const gKPI=gdeltToKPI(g);
  const cr=CREDIT_RATINGS[iso]||{};
  return{iso,country:N2[iso]||iso,CountryScore:cs,RL:cs>=65?'low':cs>=40?'medium':'high',isConflict:isCon,
    PoliticalStability:PS,Governance:GV,Corruption:COR,RuleOfLaw:RL,Voice:VA,InstitutionalCapacity:RQ,
    CivilLiberties:CL,HumanCapital:LIT,WorkingConditions:WC,SocialVulnerability:POV,Social:SOC,
    Financial:FN,Infrastructure:EL,Environmental:EN,BiodiversityAndDeforestation:BIO,EnvironmentalVulnerability:EV,
    CreditRating:CR,SP:cr.SP||null,Moody:cr.Moody||null,Fitch:cr.Fitch||null,
    GdeltRaw:g,GdeltKPI:gKPI,srcCPI:CPI[iso]!=null?'TI CPI 2024':'WGI CC',srcFH:FH[iso]!=null?'FH 2025':'—',
    wbLive:false};
}

function drawSparkline(pts,w,h){
  if(!pts||pts.length<2)return'<span style="font-size:11px;color:#94a3b8">Sin datos históricos</span>';
  const vals=pts.map(p=>p.kpi),mn=Math.min(...vals),mx=Math.max(...vals),range=mx-mn||1;
  const px=i=>Math.round((i/(pts.length-1))*w),py=v=>Math.round(h-((v-mn)/range)*(h-8)-4);
  const points=pts.map((p,i)=>`${px(i)},${py(p.kpi)}`).join(' ');
  const last=vals[vals.length-1],lx=px(pts.length-1),ly=py(last);
  const col=last>=80?'#22c55e':last>=60?'#eab308':last>=40?'#f97316':'#dc2626';
  const trend=vals[vals.length-1]-vals[0];
  const tStr=trend>2?'↑ Mejorando':trend<-2?'↓ Empeorando':'→ Estable';
  const tCol=trend>2?'#166534':trend<-2?'#991b1b':'#6b7280';
  return`<div style="display:flex;align-items:flex-end;gap:12px;flex-wrap:wrap">
    <svg width="${w}" height="${h}" style="overflow:visible;flex-shrink:0">
      <defs><linearGradient id="sg" x1="0" y1="0" x2="0" y2="1"><stop offset="0%" stop-color="${col}" stop-opacity="0.3"/><stop offset="100%" stop-color="${col}" stop-opacity="0.02"/></linearGradient></defs>
      <polyline points="${points}" fill="none" stroke="${col}" stroke-width="2.5" stroke-linecap="round" stroke-linejoin="round"/>
      <polygon points="0,${h} ${points} ${lx},${h}" fill="url(#sg)"/>
      <circle cx="${lx}" cy="${ly}" r="4" fill="${col}" stroke="#fff" stroke-width="1.5"/>
      <text x="${lx+6}" y="${ly+4}" font-size="10" fill="${col}" font-weight="700">${last}</text>
    </svg>
    <div>
      <div style="font-size:18px;font-weight:800;color:${col}">${last}<span style="font-size:11px;font-weight:400;color:#9ca3af">/100</span></div>
      <div style="font-size:11px;color:${tCol};font-weight:600">${tStr} (14 días)</div>
      <div style="font-size:10px;color:#94a3b8;margin-top:2px">${pts[0]?.date?.slice(0,8)||''} → hoy</div>
    </div>
  </div>`;
}
